/** * mincraf 1.js * ─────────────────────────────────────────────────────────── * Customizable Minecraft AI Bot — Multi-Bot Command Center * Controls up to 20 bots from one web dashboard, all in one file. * * Setup: * npm install mineflayer mineflayer-pathfinder ws * node "mincraf 1.js" * open http://localhost:3000 * * Customize bots in the BOTS array below. Shared defaults live in * DEFAULTS — anything you don't override per-bot falls back to those. * ─────────────────────────────────────────────────────────── */ // ════════════════════════════════════════════════════════════ // DEFAULTS — applied to every bot unless overridden below // ════════════════════════════════════════════════════════════ const DEFAULTS = { server: { host: "localhost", port: 25565, version: "1.20.1", // or false to auto-detect }, auth: "offline", // "microsoft" | "mojang" | "offline" password: "", owner: "YourMinecraftUsername", ownerOnly: true, commandPrefix: "!", autoReconnect: true, maxReconnectAttempts: 5, reconnectDelay: 5, // seconds behaviors: { greeting: "Hello! I'm online and ready.", deathMessage: "I died... respawning!", lowHealthThreshold: 5, lowHealthMessage: "I'm almost dead, help!", antiAfk: true, antiAfkInterval: 30000, autoLook: false, autoRespond: { "hello": "Hey there!", "bye": "Goodbye!", }, }, }; // ════════════════════════════════════════════════════════════ // BOTS — add up to 20 entries. Each only needs a username; // everything else inherits from DEFAULTS unless overridden. // ════════════════════════════════════════════════════════════ const BOTS = [ { username: "Bot01" }, { username: "Bot02" }, { username: "Bot03" }, // { username: "Bot04" }, // { username: "Bot05", server: { host: "other.server.net", port: 25565 } }, // ...add up to 20 total ]; const DASHBOARD_PORT = 3000; // ════════════════════════════════════════════════════════════ // DEPENDENCIES // ════════════════════════════════════════════════════════════ const mineflayer = require('mineflayer'); const { pathfinder, Movements, goals } = require('mineflayer-pathfinder'); const http = require('http'); const { WebSocketServer } = require('ws'); if (BOTS.length === 0) { console.error('No bots configured. Add at least one entry to the BOTS array.'); process.exit(1); } if (BOTS.length > 20) { console.error('Maximum of 20 bots supported. Trim the BOTS array.'); process.exit(1); } function merge(base, override) { const out = { ...base, ...override }; for (const key of ['server', 'behaviors']) { out[key] = { ...(base[key] || {}), ...(override[key] || {}) }; } out.behaviors.autoRespond = { ...(base.behaviors?.autoRespond || {}), ...(override.behaviors?.autoRespond || {}) }; return out; } // Each entry: { id, cfg, bot, reconnectAttempts, statePushInterval, intervals:{} } const instances = new Map(); BOTS.forEach((override, idx) => { const id = override.username || `bot${idx + 1}`; const cfg = merge(DEFAULTS, override); instances.set(id, { id, cfg, bot: null, reconnectAttempts: 0, statePushInterval: null, antiAfkInterval: null, autoLookInterval: null, }); }); // ════════════════════════════════════════════════════════════ // DASHBOARD SERVER // ════════════════════════════════════════════════════════════ let wsClients = new Set(); function startDashboard(port) { const server = http.createServer((req, res) => { res.writeHead(200, { 'Content-Type': 'text/html' }); res.end(DASHBOARD_HTML); }); const wss = new WebSocketServer({ server }); wss.on('connection', (ws) => { wsClients.add(ws); // send a full snapshot immediately so a freshly-opened tab is in sync for (const inst of instances.values()) { ws.send(JSON.stringify({ type: 'state', id: inst.id, state: getState(inst) })); } ws.on('message', (raw) => { try { const msg = JSON.parse(raw.toString()); if (msg.type === 'command') { if (msg.id === '__all__') { for (const inst of instances.values()) { const result = runCommand(inst, msg.cmd, msg.args || []); broadcastLog(inst.id, `> ${result}`, 'cmd'); } } else { const inst = instances.get(msg.id); if (inst) { const result = runCommand(inst, msg.cmd, msg.args || []); broadcastLog(inst.id, `> ${result}`, 'cmd'); } } } } catch (e) { /* ignore malformed */ } }); ws.on('close', () => wsClients.delete(ws)); }); server.listen(port, () => { console.log(`\n🖥️ Command Center running at http://localhost:${port}\n`); }); } function broadcastState(id, state) { const payload = JSON.stringify({ type: 'state', id, state }); for (const ws of wsClients) if (ws.readyState === ws.OPEN) ws.send(payload); } function broadcastLog(id, text, kind = 'info') { const payload = JSON.stringify({ type: 'log', id, text, kind }); for (const ws of wsClients) if (ws.readyState === ws.OPEN) ws.send(payload); } // ════════════════════════════════════════════════════════════ // LOGGING + STATE // ════════════════════════════════════════════════════════════ function log(id, msg) { const ts = new Date().toLocaleTimeString(); console.log(`[${ts}] [${id}] ${msg}`); broadcastLog(id, `[${ts}] ${msg}`); } function getState(inst) { const bot = inst.bot; if (!bot || !bot.entity) return { connected: false, username: inst.id }; return { connected: true, username: bot.username, health: bot.health, food: bot.food, position: bot.entity.position, dimension: bot.game?.dimension, time: bot.time?.timeOfDay, players: Object.keys(bot.players), inventory: bot.inventory.items().map(i => ({ name: i.name, count: i.count })), }; } function pushState(inst) { broadcastState(inst.id, getState(inst)); } // ════════════════════════════════════════════════════════════ // BOT LIFECYCLE // ════════════════════════════════════════════════════════════ function createBot(inst) { const { cfg, id } = inst; log(id, `Connecting to ${cfg.server.host}:${cfg.server.port} as ${id}...`); const bot = mineflayer.createBot({ host: cfg.server.host, port: cfg.server.port, username: id, password: cfg.password || undefined, version: cfg.server.version || false, auth: cfg.auth || 'offline', }); inst.bot = bot; bot.loadPlugin(pathfinder); bot.once('spawn', () => { inst.reconnectAttempts = 0; log(id, `✅ Spawned in world`); bot.pathfinder.setMovements(new Movements(bot)); if (cfg.behaviors.greeting) { setTimeout(() => bot.chat(cfg.behaviors.greeting), 1500); } if (cfg.behaviors.autoLook) startAutoLook(inst); if (cfg.behaviors.antiAfk) startAntiAfk(inst); pushState(inst); clearInterval(inst.statePushInterval); inst.statePushInterval = setInterval(() => pushState(inst), 1000); }); bot.on('chat', (username, message) => { if (username === bot.username) return; log(id, `[CHAT] ${username}: ${message}`); broadcastLog(id, `💬 ${username}: ${message}`, 'chat'); handleChat(inst, username, message); }); bot.on('health', () => { pushState(inst); if (bot.health <= cfg.behaviors.lowHealthThreshold && cfg.behaviors.lowHealthMessage) { bot.chat(cfg.behaviors.lowHealthMessage); } }); bot.on('death', () => { log(id, '💀 Died.'); pushState(inst); if (cfg.behaviors.deathMessage) bot.chat(cfg.behaviors.deathMessage); }); bot.on('kicked', (reason) => { log(id, `⚠️ Kicked: ${reason}`); teardown(inst); tryReconnect(inst); }); bot.on('error', (err) => log(id, `❌ Error: ${err.message}`)); bot.on('end', () => { log(id, '🔌 Disconnected.'); teardown(inst); tryReconnect(inst); }); } function teardown(inst) { clearInterval(inst.statePushInterval); clearInterval(inst.antiAfkInterval); clearInterval(inst.autoLookInterval); broadcastState(inst.id, { connected: false, username: inst.id }); } function startAntiAfk(inst) { const { bot, cfg } = inst; clearInterval(inst.antiAfkInterval); inst.antiAfkInterval = setInterval(() => { if (!bot.entity) return; const moves = ['forward', 'back', 'left', 'right']; const dir = moves[Math.floor(Math.random() * moves.length)]; bot.setControlState(dir, true); setTimeout(() => bot.setControlState(dir, false), 500); bot.look(Math.random() * Math.PI * 2, (Math.random() - 0.5) * 0.5, false); }, cfg.behaviors.antiAfkInterval || 30000); } function startAutoLook(inst) { const { bot } = inst; clearInterval(inst.autoLookInterval); inst.autoLookInterval = setInterval(() => { if (!bot.entity) return; const players = Object.values(bot.players).filter(p => p.entity && p.username !== bot.username); if (players.length > 0) bot.lookAt(players[0].entity.position.offset(0, 1.6, 0)); }, 2000); } function handleChat(inst, username, message) { const { cfg, bot } = inst; const owner = cfg.owner; const isOwner = !owner || username === owner; const prefix = cfg.commandPrefix || '!'; if (!message.startsWith(prefix)) { if (cfg.behaviors.autoRespond) { for (const [trigger, response] of Object.entries(cfg.behaviors.autoRespond)) { if (message.toLowerCase().includes(trigger.toLowerCase())) { setTimeout(() => bot.chat(response), 500); return; } } } return; } const args = message.slice(prefix.length).trim().split(/\s+/); const cmd = args.shift().toLowerCase(); if (!isOwner && cfg.ownerOnly) { bot.chat(`Only ${owner} can command me.`); return; } runCommand(inst, cmd, args, username); } // ════════════════════════════════════════════════════════════ // COMMANDS — shared by in-game chat and the dashboard // ════════════════════════════════════════════════════════════ function runCommand(inst, cmd, args = [], sourceUsername = null) { const bot = inst.bot; if (!bot || !bot.entity) return `${inst.id}: not connected.`; const say = (msg) => { bot.chat(msg); return msg; }; switch (cmd) { case 'come': case 'follow': { const targetName = sourceUsername || args[0]; const target = targetName ? bot.players[targetName]?.entity : null; if (!target) return say("I can't see that player!"); bot.pathfinder.setGoal(new goals.GoalFollow(target, 2), true); return say(`Following ${targetName}!`); } case 'stop': bot.pathfinder.stop(); bot.clearControlStates(); return say('Stopped.'); case 'goto': { const [x, y, z] = args.map(Number); if ([x, y, z].some(isNaN)) return 'Usage: goto '; bot.pathfinder.setGoal(new goals.GoalBlock(x, y, z)); return say(`Heading to ${x} ${y} ${z}…`); } case 'say': return args.length ? say(args.join(' ')) : 'Nothing to say.'; case 'pos': { const p = bot.entity.position; return say(`I'm at ${Math.floor(p.x)}, ${Math.floor(p.y)}, ${Math.floor(p.z)}`); } case 'health': return say(`❤ Health: ${bot.health}/20 🍗 Food: ${bot.food}/20`); case 'time': return say(`In-game time: ${bot.time.timeOfDay} (day ${bot.time.day})`); case 'inventory': { const items = bot.inventory.items().map(i => `${i.name}×${i.count}`).join(', ') || 'empty'; return say(`Inventory: ${items}`); } case 'look': { const [lx, ly, lz] = args.map(Number); if ([lx, ly, lz].some(isNaN)) return 'Usage: look '; bot.lookAt({ x: lx, y: ly, z: lz }); return `Looking at ${lx} ${ly} ${lz}`; } case 'jump': bot.setControlState('jump', true); setTimeout(() => bot.setControlState('jump', false), 500); return 'Jumped.'; case 'sneak': bot.setControlState('sneak', !bot.getControlState('sneak')); return say(`Sneak: ${bot.getControlState('sneak') ? 'on' : 'off'}`); case 'attack': { const targetName = args[0]; const entity = targetName ? Object.values(bot.entities).find(e => e.username === targetName || e.name === targetName) : bot.nearestEntity(); if (!entity) return say('No target found.'); bot.attack(entity); return say(`Attacking ${entity.username || entity.name || 'entity'}!`); } case 'forward': case 'back': case 'left': case 'right': bot.pathfinder.stop(); bot.setControlState(cmd, true); setTimeout(() => bot.setControlState(cmd, false), 800); return `Moving ${cmd}.`; case 'drop': { const item = bot.inventory.items()[0]; if (!item) return 'Nothing to drop.'; bot.tossStack(item).catch(() => {}); return say(`Dropped ${item.name}.`); } case 'disconnect': bot.quit(); return 'Disconnecting…'; case 'reconnect': inst.reconnectAttempts = 0; bot.quit(); setTimeout(() => createBot(inst), 1000); return 'Reconnecting…'; case 'help': return say('Commands: come stop goto say pos health inventory attack jump sneak time forward/back/left/right drop'); default: return `Unknown command: ${cmd}`; } } function tryReconnect(inst) { const cfg = inst.cfg; const max = cfg.autoReconnect ? (cfg.maxReconnectAttempts || 5) : 0; if (inst.reconnectAttempts >= max) { if (max > 0) log(inst.id, 'Max reconnect attempts reached. Giving up.'); return; } inst.reconnectAttempts++; const delay = (cfg.reconnectDelay || 5) * 1000; log(inst.id, `Reconnecting in ${delay / 1000}s… (attempt ${inst.reconnectAttempts}/${max})`); setTimeout(() => createBot(inst), delay); } // ════════════════════════════════════════════════════════════ // EMBEDDED DASHBOARD HTML (multi-bot command center) // ════════════════════════════════════════════════════════════ const DASHBOARD_HTML = ` Command Center

COMMAND CENTER

connecting…
0 online 0 offline

FLEET (0)

Select a bot from the fleet list, or enable broadcast mode to control all bots at once.
`; // ════════════════════════════════════════════════════════════ // START // ════════════════════════════════════════════════════════════ startDashboard(DASHBOARD_PORT); for (const inst of instances.values()) { createBot(inst); }
